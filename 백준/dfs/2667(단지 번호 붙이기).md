### 단지 번호 붙이기
  - 난이도 실버 1  
  - [단지 번호 붙이기 풀러가기](https://www.acmicpc.net/problem/2667)
  - java code
  ```java
    import java.io.BufferedReader;
    import java.io.InputStreamReader;
    import java.util.PriorityQueue;
    
    public class Main {
    
        static int N, blocks, numberOfHouseholds;
        static int[][] cities;
        static PriorityQueue<Integer> priorityQueue = new PriorityQueue<>();
        static int[][] d = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        public static void main(String[] args) throws Exception {
            BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
            N = Integer.parseInt(br.readLine());
    
            cities = new int[N][N];
            for (int i = 0; i < N; i++) {
                String cityLine = br.readLine();
                for (int j = 0; j < N; j++) {
                    cities[i][j] = cityLine.charAt(j) - '0';
                }
            }
    
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < N; j++) {
                    if (cities[i][j] == 1) {
                        numberOfHouseholds = 0;
                        dfs(i, j);
                        blocks += 1;
                        priorityQueue.offer(numberOfHouseholds == 0 ? 1 : numberOfHouseholds);
                    }
                }
            }
    
            System.out.println(blocks);
            StringBuilder sb = new StringBuilder();
            while (!priorityQueue.isEmpty()) {
                sb.append(priorityQueue.poll()).append("\n");
            }
            System.out.println(sb);
        }
    
        private static void dfs(int row, int col) {
            for (int i = 0; i < d.length; i++) {
                int nextRow = row + d[i][0];
                int nextCol = col + d[i][1];
    
                if (nextRow >= N || nextRow < 0 || nextCol >= N || nextCol < 0) continue;
                if (cities[nextRow][nextCol] != 1) continue;
                cities[nextRow][nextCol] = 2;
                numberOfHouseholds += 1;
                dfs(nextRow, nextCol);
            }
        }
    }
  ```

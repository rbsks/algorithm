### 치킨 배달
  - 난이도 골드 5  
  - [치킨 배달 풀러가기](https://www.acmicpc.net/problem/156866)
  - java code
  ```java
    import java.io.BufferedReader;
    import java.io.InputStreamReader;
    import java.util.*;
    
    public class Main {
    
        static int N, M, result = Integer.MAX_VALUE;
        static ArrayList<int[]> house = new ArrayList<>();
        static ArrayList<int[]> chicken = new ArrayList<>();
        static ArrayList<int[]> choice = new ArrayList<>();
        static boolean[] visited;
        public static void main(String[] args) throws Exception {
            BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    
            StringTokenizer nm = new StringTokenizer(br.readLine(), " ");
            N = Integer.parseInt(nm.nextToken());
            M = Integer.parseInt(nm.nextToken());
    
            for (int i = 0; i < N; i++) {
                StringTokenizer city = new StringTokenizer(br.readLine(), " ");
                for (int j = 0; j < N; j++) {
                    int num = Integer.parseInt(city.nextToken());
                    if (num == 1) {
                        house.add(new int[]{i, j});
                    } else if (num == 2) {
                        chicken.add(new int[]{i, j});
                    }
                }
            }
    
            visited = new boolean[chicken.size()];
    
            combination(0, 0);
            System.out.println(result);
        }
    
        public static void combination(int depth, int start) {
            if(depth == M) {
                int sum = 0;
                for (int[] ints : house) {
                    int min = Integer.MAX_VALUE;
                    for (int[] ints1 : choice) {
                        min = Math.min(min, Math.abs(ints[0] - ints1[0]) + Math.abs(ints[1] - ints1[1]));
                    }
                    sum += min;
                }
                result = Math.min(result, sum);
                return;
            }
    
            for(int i = start; i < chicken.size(); i++) {
                if (!visited[i]) {
                    visited[i] = true;
                    choice.add(chicken.get(i));
                    combination(depth + 1, i + 1);
                    choice.remove(choice.size() - 1);
                    visited[i] = false;
                }
            }
        }
    }
  ```
